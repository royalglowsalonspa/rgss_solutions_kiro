5. Caching Implementation

5.1 Cloudflare KV Cache Pattern

// Read-through cache for service catalog
async function getServices(branchId: string) {
  const cacheKey = `services:${branchId}`

  // L1: Check KV edge cache
  const cached = await KV.get(cacheKey, 'json')
  if (cached) return cached as ServiceWithCategory[]

  // L2: Query database (source of truth)
  const services = await db.query.service.findMany({
    where: eq(service.branchId, branchId),
    with: { category: true, serviceStaff: { with: { staff: true } } },
    orderBy: [asc(service.sortOrder)]
  })

  // Write to cache (5-min TTL)
  await KV.put(cacheKey, JSON.stringify(services), { expirationTtl: 300 })
  return services
}

5.2 Redis Slot Availability Cache

// Key: availability:{branch}:{date}  |  TTL: 5 minutes
// Invalidation: on booking confirm/cancel → DELETE key

async function getAvailableSlots(branchId: string, date: string) {
  const cacheKey = `availability:${branchId}:${date}`

  const cached = await redis.get<SlotData[]>(cacheKey)
  if (cached) return cached

  const allSlots = await calculateAvailability(branchId, date)
  await redis.set(cacheKey, allSlots, { ex: 300 })
  return allSlots
}

// Invalidation on booking state change:
async function invalidateSlotCache(branchId: string, date: string) {
  await redis.del(`availability:${branchId}:${date}`)
}

6. Realtime Implementation

6.1 Token Auth Flow

// POST /api/ably/token
export async function POST(req: Request) {
  // 1. Validate session
  const session = await getSession(req)
  if (!session) return Response.json({ error: 'Unauthorized' }, { status: 401 })

  // 2. Determine capabilities by role
  const capabilities = getCapabilitiesForRole(session.user.role, session.user.id)
  // Customer:     customer:{userId}:bookings → [subscribe]
  // Staff:        + staff:{staffId}:schedule → [subscribe]
  // Admin roles:  + admin:bookings, admin:schedule:*, admin:leave

  // 3. Generate time-limited token (1 hour TTL)
  const tokenRequest = await ably.auth.createTokenRequest({
    clientId: session.user.id,
    capability: capabilities,
    ttl: 3600 * 1000,
  })

  return Response.json(tokenRequest)
}

8. Invoice & GST Calculation

8.1 GST Back-Calculation

const GST_RATE = 0.18           // 18% total (CGST 9% + SGST 9%)
const SAC_CODE = '999721'       // Beauty & wellness services

interface GSTBreakdown {
  basePaise: number            // Pre-tax amount
  cgstPaise: number            // Central GST (9%)
  sgstPaise: number            // State GST (9%)
  totalGstPaise: number        // Total GST (18%)
  totalPaise: number           // Original inclusive amount
}

export function splitGST(inclusivePaise: number): GSTBreakdown {
  const basePaise = Math.round(inclusivePaise / (1 + GST_RATE))
  const totalGstPaise = inclusivePaise - basePaise
  const cgstPaise = Math.round(totalGstPaise / 2)
  const sgstPaise = totalGstPaise - cgstPaise  // Avoids rounding mismatch
  return { basePaise, cgstPaise, sgstPaise, totalGstPaise, totalPaise: inclusivePaise }
}

// splitGST(118000) → base: ₹1,000 | CGST: ₹90 | SGST: ₹90 | Total: ₹1,180

8.3 Booking Number Generation

// Format: BK{BRANCH_CODE}{DDMM}{SERVICE_TYPE_INITIAL}{5-CHAR-NANOID}
// Example: BKRS2605S38a9k

const BRANCH_CODES = { branch_rajajinagar: 'RS', branch_malleshwaram: 'MS' }
const TYPE_INITIALS = { salon: 'S', spa: 'P' }

export function generateBookingNumber(branchId: string, serviceType: string) {
  const branch = BRANCH_CODES[branchId] || 'XX'
  const dd = String(new Date().getDate()).padStart(2, '0')
  const mm = String(new Date().getMonth() + 1).padStart(2, '0')
  const type = TYPE_INITIALS[serviceType] || 'X'
  return `BK${branch}${dd}${mm}${type}${nanoid(5)}`
}
